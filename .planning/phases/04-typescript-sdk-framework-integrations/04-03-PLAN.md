Python (Phase 2 Plan 01 — already shipped):
```python
from wbauth import Identity, attach_signing, WebBotAuth, WebBotAuthAdapter

identity = Identity.load_or_generate(
    Path("~/.config/wbauth/key.pem").expanduser(),
    signature_agent_url="https://...",
    user_agent="my-bot/1.0",  # optional
)
identity.kid  # str — RFC 7638 thumbprint

# httpx adapter (used by openai_agents_demo)
import httpx
with httpx.Client(auth=WebBotAuth(identity)) as client:
    resp = client.get("https://example.com")  # signed

# Playwright adapter (used by browser_use_demo)
from playwright.async_api import async_playwright
async with async_playwright() as p:
    browser = await p.chromium.launch()
    page = await browser.new_page()
    await attach_signing(page, identity)  # MUST be before goto (Pitfall 6)
    await page.goto("https://example.com")
```
