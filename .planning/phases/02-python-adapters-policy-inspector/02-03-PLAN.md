       ```python
       def _build_parser():
           parser = argparse.ArgumentParser(prog="wbauth", ...)
           sub = parser.add_subparsers(dest="cmd", required=True)
           # ... existing keygen ...
           insp = sub.add_parser("inspect",
               help="Run pre-flight policy inspector against URL")
           insp.add_argument("url", help="Target URL (e.g., https://example.com)")
           insp.add_argument("--json", action="store_true",
               help="Emit machine-readable JSON to stdout")
           # verify in Task 3
           return parser

       def _dispatch_inspect(args) -> int:
           import asyncio, json, sys
           from .policy import inspect, SitePolicy
           try:
               result: SitePolicy = asyncio.run(inspect(args.url))
           except ValueError as e:
               print(f"error: invalid URL: {e}", file=sys.stderr)
               return 3
           except Exception as e:
               print(f"error: {type(e).__name__}: {e}", file=sys.stderr)
               return 3
           if args.json:
               print(json.dumps(_serialize_policy(result), default=str))
           else:
               _print_human_summary(result)
           return {"allowed": 0, "restricted": 1, "forbidden": 2}.get(result.verdict, 3)

       def _serialize_policy(p):
           # dataclasses.asdict handles nested dataclasses; datetime via default=str.
           # Exception objects in p.errors are not dataclasses → custom dict.
           import dataclasses
           d = dataclasses.asdict(p)
           d["errors"] = {
               name: {"type": type(exc).__name__, "message": str(exc)}
               for name, exc in p.errors.items()
           }
           return d

       def _print_human_summary(p):
           print(f"Verdict: {p.verdict}")
           print(f"URL:     {p.url}")
           print("Reasons:")
           for r in p.reasons:
               print(f"  - {r}")
           print(f"Partial: {str(p.partial).lower()}")
           if p.errors:
               print("Errors:")
               for name, exc in p.errors.items():
                   print(f"  - {name}: {type(exc).__name__}: {exc}")
           else:
               print("Errors:  none")
           print(f"Fetched: {p.fetched_at.isoformat()}")
           print()
           print("(For full SitePolicy JSON, re-run with --json)")
       ```
