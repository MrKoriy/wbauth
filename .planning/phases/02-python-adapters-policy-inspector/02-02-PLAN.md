       ```python
                       f"llms.txt description suggests no automated access: {m.group(0)!r}"

           if signing_directory is not None and signing_directory.present:
       ```

       ```python
       Parallel fan-out to 4 well-known endpoints with per-task timeout (3s),
       isolated failure handling (return_exceptions=True), strict verdict per
       D-18, in-process per-host LRU cache per D-22.
       """
       from __future__ import annotations
       import asyncio
       import datetime
       from urllib.parse import urlsplit

       import httpx

       from .cache import PolicyCache
       from .errors import RobotsParseError
       from .parsers import (
           parse_robots, parse_ai_txt, parse_llms_txt, parse_signing_directory,
       )
       from .policy import SitePolicy
       from .verdict import compute_verdict

       PER_ENDPOINT_TIMEOUT = 3.0  # POLICY-02
       USER_AGENT = "wbauth/0.1"   # D-20
       MAX_REDIRECTS = 3           # Open question #1

       _CACHE = PolicyCache()  # module-level singleton, per D-23


       def _reset_cache_for_tests() -> None:
           """Test-only: clear cache between cases. Not part of public API."""
           global _CACHE
           _CACHE = PolicyCache()


       def _endpoints(host: str) -> dict[str, str]:
           return {
               "robots": f"https://{host}/robots.txt",
               "ai_txt": f"https://{host}/ai.txt",
               "llms_txt": f"https://{host}/llms.txt",
               "signing_directory": f"https://{host}/.well-known/http-message-signatures-directory",
           }


       async def _fetch_one(client: httpx.AsyncClient, url: str) -> httpx.Response:
           response = await client.get(
               url,
               follow_redirects=True,
               max_redirects=MAX_REDIRECTS,
               headers={"User-Agent": USER_AGENT},
           )
           response.raise_for_status()
           return response


       async def _fetch_missing(host: str, missing: list[str]) -> dict[str, httpx.Response | Exception]:
           if not missing:
               return {}
           urls = _endpoints(host)
           async with httpx.AsyncClient(timeout=PER_ENDPOINT_TIMEOUT) as client:
               coros = [
                   asyncio.wait_for(_fetch_one(client, urls[name]), timeout=PER_ENDPOINT_TIMEOUT)
                   for name in missing
               ]
               results = await asyncio.gather(*coros, return_exceptions=True)
           return dict(zip(missing, results, strict=True))


       def _parse(name: str, response: httpx.Response, target_url: str):
           ct = response.headers.get("content-type")
           if name == "robots":
               return parse_robots(response.text, ct, target_url, USER_AGENT)
           if name == "ai_txt":
               return parse_ai_txt(response.text)
           if name == "llms_txt":
               return parse_llms_txt(response.text)
           if name == "signing_directory":
               return parse_signing_directory(response.text, ct)
           raise ValueError(f"unknown endpoint {name!r}")  # defensive


       async def inspect(url: str) -> SitePolicy:
           """Run the pre-flight policy inspector for `url` and return SitePolicy.

           - 4 parallel HTTPS GETs (3s each) per POLICY-02
           - Robots evaluated against the input URL's path with UA 'wbauth/0.1'
             (open question #2 resolution)
           - Strict verdict per D-18 (no identity-related signals → ambiguity = restricted)
           - Per-(host, endpoint) cache hits avoid re-fetching (D-22)
           - Zero HTTP calls to wbauth-controlled domains (POLICY-08)
           """
           parsed = urlsplit(url)
           host = parsed.netloc
           if not host:
               raise ValueError(f"inspect: URL has no host: {url!r}")

           # 1. Cache lookup
           cached: dict[str, object] = {}
           missing: list[str] = []
           for ep in ("robots", "ai_txt", "llms_txt", "signing_directory"):
               entry = _CACHE.get(host, ep)
               if entry is not None:
                   cached[ep] = entry.value
               else:
                   missing.append(ep)

           # 2. Fetch missing in parallel
           fetched = await _fetch_missing(host, missing)

           results: dict[str, object | None] = {
               "robots": None, "ai_txt": None, "llms_txt": None, "signing_directory": None,
           }
           errors: dict[str, Exception] = {}

           # Restore cached
           for ep, value in cached.items():
               results[ep] = value

           # Parse newly fetched
           for ep, fr in fetched.items():
               if isinstance(fr, Exception):
                   errors[ep] = fr
                   continue
               try:
                   parsed_value = _parse(ep, fr, url)
                   results[ep] = parsed_value
                   _CACHE.set(host, ep, parsed_value,
                              cache_control=fr.headers.get("cache-control"),
                              etag=fr.headers.get("etag"))
               except RobotsParseError as e:
                   errors[ep] = e   # let verdict engine handle
               except Exception as e:
                   errors[ep] = e

           partial = bool(errors)
           verdict, reasons = compute_verdict(
               robots=results["robots"], ai_txt=results["ai_txt"],
               llms_txt=results["llms_txt"], signing_directory=results["signing_directory"],
               errors=errors, partial=partial,
           )
           return SitePolicy(
               url=url,
               robots=results["robots"], ai_txt=results["ai_txt"],
               llms_txt=results["llms_txt"], signing_directory=results["signing_directory"],
               verdict=verdict, reasons=reasons, partial=partial, errors=errors,
               fetched_at=datetime.datetime.now(datetime.timezone.utc),
           )
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user agent process → user-supplied origin | `inspect(url)` makes 4 outbound HTTPS GETs to the user-supplied origin's well-known endpoints. The origin is not under our control; its responses are untrusted input. |
| origin response body → parsers | parsers consume raw bytes from a third-party origin and produce structured Result objects; malicious payloads cross this boundary. |
| `inspect()` result → caller | the caller (CLI, library user, demo) consumes `SitePolicy.verdict` and `reasons` to decide whether to scrape; bad parsing → bad verdict → bad agent decision. |
| `inspect()` → wbauth-controlled domains | INTENTIONALLY EMPTY per POLICY-08. Any HTTP call to wbauth.dev / agentpassport.dev would breach the zero-cloud-dependency invariant. |

Body-size cap is documented as a v1.x add (`httpx.AsyncClient(limits=...)`); for v1, the timeout is the deterministic bound. Threat surface ≈ 3s × 4 endpoints = 12s worst-case `inspect()`. |
