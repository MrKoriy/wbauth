       ```python
       """httpx.Auth adapter for Web Bot Auth (ADAPT-01).

       Delegates all crypto to wbauth.sign() per L-02. Stateless per D-15.
       """
       from __future__ import annotations
       import httpx
       from wbauth import NormalizedRequest, sign


       class WebBotAuth(httpx.Auth):
           """Drop-in httpx Auth that signs every request via wbauth.sign().

           Usage:
               httpx.Client(auth=WebBotAuth(identity)).get(url)
               httpx.AsyncClient(auth=WebBotAuth(identity)).get(url)

           If `identity.user_agent` is set AND the outgoing request has no
           User-Agent header, the adapter sets it. Otherwise UA is untouched.
           """
           # Required: read the body before auth_flow so content-digest is stable.
           requires_request_body = True

           def __init__(self, identity):
               self._identity = identity

           def sync_auth_flow(self, request):
               self._sign(request)
               yield request

           async def async_auth_flow(self, request):
               # No I/O during signing — async path is identical (verified per
               # Context7 /encode/httpx: "auth classes designed to not perform I/O").
               self._sign(request)
               yield request

           def _sign(self, request):
               body = request.content if request.content else None
               normalized = NormalizedRequest(
                   method=request.method,
                   url=str(request.url),
                   headers=dict(request.headers),
                   body=body,
               )
               sig = sign(normalized, self._identity)
               request.headers["Signature"] = sig.signature
               request.headers["Signature-Input"] = sig.signature_input
               request.headers["Signature-Agent"] = sig.signature_agent
               # Open question #4 resolution: identity.user_agent → request UA
               # only when the outgoing request has no UA already.
               ua = getattr(self._identity, "user_agent", None)
               if ua and "user-agent" not in {k.lower() for k in request.headers}:
                   request.headers["User-Agent"] = ua
       ```

    2. **Replace** the Task 1 stub at `python/src/wbauth/adapters/requests_adapter.py` with the production implementation, mirroring §"Pattern 2" + UA injection:

       ```python
       """requests.auth.AuthBase adapter for Web Bot Auth (ADAPT-02).

       Delegates all crypto to wbauth.sign() per L-02. Stateless per D-15.
       """
       from __future__ import annotations
       from requests.auth import AuthBase
       from wbauth import NormalizedRequest, sign


       class WebBotAuthAdapter(AuthBase):
           """Drop-in requests Auth that signs every request via wbauth.sign().

           Usage:
               session = requests.Session()
               session.get(url, auth=WebBotAuthAdapter(identity))
           """
           def __init__(self, identity):
               self._identity = identity

           def __call__(self, prepared_request):
               body = prepared_request.body
               if isinstance(body, str):
                   body = body.encode("utf-8")
               normalized = NormalizedRequest(
                   method=prepared_request.method,
                   url=prepared_request.url,
                   headers=dict(prepared_request.headers),
                   body=body,
               )
               sig = sign(normalized, self._identity)
               prepared_request.headers["Signature"] = sig.signature
               prepared_request.headers["Signature-Input"] = sig.signature_input
               prepared_request.headers["Signature-Agent"] = sig.signature_agent
               ua = getattr(self._identity, "user_agent", None)
               if ua and "User-Agent" not in prepared_request.headers:
                   prepared_request.headers["User-Agent"] = ua
               return prepared_request
       ```

       ```python
       """Adapter conformance against spec/test-vectors/ — proves byte-equality
       with the Phase-1 signer baseline (D-29, ADAPT-06)."""
       import json, datetime, pathlib
       from base64 import urlsafe_b64decode
       import httpx
       from cryptography.hazmat.primitives.asymmetric.ed25519 import Ed25519PrivateKey
       from wbauth import Identity, KeyPair
       from wbauth.identity import _compute_kid  # private but stable
       from wbauth.adapters import WebBotAuth, WebBotAuthAdapter

       VECTORS = pathlib.Path(__file__).parents[2] / "spec" / "test-vectors"

       def _identity_from_vector(vector_dir: pathlib.Path) -> Identity:
           data = json.loads((vector_dir / "input.json").read_text())
           jwk = data["identity"]["private_key_jwk"]
           # base64url decode the OKP `d` to 32 raw bytes:
           d_bytes = urlsafe_b64decode(jwk["d"] + "==")
           priv = Ed25519PrivateKey.from_private_bytes(d_bytes)
           kp = KeyPair(priv, _compute_kid(priv.public_key()))
           return Identity(kp, signature_agent_url=data["identity"]["signature_agent_url"])

       def _sign_via_httpx_adapter(vector_dir: pathlib.Path) -> dict[str, str]:
           data = json.loads((vector_dir / "input.json").read_text())
           identity = _identity_from_vector(vector_dir)
           captured: list[httpx.Request] = []
           def handler(req):
               captured.append(req)
               return httpx.Response(200)
           # Build the request the vector describes, intercept via MockTransport.
           # Override the default created/nonce/label on sign() by monkeypatching;
           # easier: call sign() directly with vector params, but the test's job
           # is to prove the *adapter path* produces these bytes — so we need a
           # way to freeze created/nonce. Approach: monkeypatch wbauth.signer.sign
           # to wrap the real sign() with vector-fixed kwargs for this one test.
           import wbauth.signer as _sm
           real = _sm.sign
           sp = data["signing_params"]
           created_dt = datetime.datetime.fromtimestamp(sp["created"], tz=datetime.timezone.utc)
           def _patched(req, ident, **kw):
               return real(req, ident, created=created_dt, nonce=sp["nonce"],
                           label=sp["label"], expires_after_seconds=sp["expires_after_seconds"])
           _sm.sign = _patched
           try:
               c = httpx.Client(auth=WebBotAuth(identity),
                                transport=httpx.MockTransport(handler))
               r = data["request"]
               c.request(r["method"], r["url"], headers=r.get("headers") or {},
                         content=r.get("body"))
           finally:
               _sm.sign = real
           sent = captured[0]
           return {
               "signature": sent.headers["Signature"],
               "signature_input": sent.headers["Signature-Input"],
               "signature_agent": sent.headers["Signature-Agent"],
           }

       Registers a page.route handler that intercepts every outgoing request,
       signs it via wbauth.sign(), and continues with the signed headers.
       Stateless per D-15; async-only per D-13.

       PITFALL 6: Call attach_signing(page, identity) BEFORE the first
       page.goto()/page.click() that should produce signed requests. page.route

       from __future__ import annotations
       from wbauth import NormalizedRequest, sign


       async def attach_signing(page, identity) -> None:
           """Register a route handler on `page` that signs every outgoing request.

           Idempotent contract: caller is responsible for calling once per page;
           registering twice yields two handlers and only the first wins (Playwright
           default for overlapping routes).
           """
           async def _handler(route, request):
               headers = dict(await request.all_headers())
               normalized = NormalizedRequest(
                   method=request.method,
                   url=request.url,
                   headers=headers,
                   body=request.post_data_buffer,
               )
               sig = sign(normalized, identity)
               headers["Signature"] = sig.signature
               headers["Signature-Input"] = sig.signature_input
               headers["Signature-Agent"] = sig.signature_agent
               ua = getattr(identity, "user_agent", None)
               if ua and "user-agent" not in {k.lower() for k in headers}:
                   headers["User-Agent"] = ua
               await route.continue_(headers=headers)
           await page.route("**/*", _handler)
       ```
